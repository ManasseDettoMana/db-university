    FROM `students` 
    GROUP BY `enrolment_date`;

2. SELECT COUNT(*) AS `numero_insegnanti`, `office_address` 
    FROM `teachers` 
    GROUP BY `office_address`;

3. SELECT AVG(`vote`), `exam_id` 
    FROM `exam_student` 
    GROUP BY `exam_id`;

4. SELECT COUNT(*), `department_id` 
    FROM `degrees` 
    GROUP BY `department_id`;
